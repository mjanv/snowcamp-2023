Pipelines

        
        defmodule Maths do
            def oneliner(n) do
                Enum.sum( Enum.map(Enum.filter(1..n, fn x -> Integer.is_odd(x) end), 
                fn x -> x * x end))
            end

            def pipeline(n) do
                1..n
                |> Enum.filter(fn x -> Integer.is_odd(x) end)
                |> Enum.map(fn x -> x * x end)
                |> Enum.sum()
            end
        end 
        
    
        
            iex> Maths.oneliner(100)
            338350
            
            iex> Maths.pipeline(100)
            338350

        defmodule Sums do
            def sum([]), do: 0.0
            def sum([%Item{price: price} | items]), do: price + sum(items)

            def sum([]), do: 0.0
            def sum([head | tail]), do: head + sum(tail)

            def sum(_), do: -1.0
        end
        
    
        
            iex> Sums.sum([%Item{price: 2.1}, %Item{price: 3.4}, %Item{price: 1.0}])
            6.5
        
            iex> Sums.sum([1.0, 2.0, 3.0])
            6.0

            iex> Sums.sum(%{a: 1})
            -1.0
